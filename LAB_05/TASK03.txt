.MODEL SMALL
.STACK 100H

.DATA

MSG1 DB "Enter an array of size 5: $"  

A DB 5 DUP(?)

.CODE
MAIN PROC

MOV AX, @DATA
MOV DS, AX

LEA DX, MSG1
MOV AH, 9H
INT 21H

MOV CX, 5
MOV BX, OFFSET A
MOV AH, 1

START:
INT 21H
MOV [BX], AL
INC BX
LOOP START

MOV CX, 5
DEC CX

THEN:
MOV BX, CX
MOV SI, 0 

AFTER:
MOV AL, A[SI]
MOV DL, A[SI+1]
CMP AL, DL
JC GO

MOV A[SI], DL
MOV A[SI+1], AL

GO:
INC SI
DEC BX
JNZ AFTER

LOOP THEN

MOV AH, 2
MOV DL, 10
INT 21H

MOV DL, 13
INT 21H

MOV CX, 5
MOV BX, OFFSET A

OUTPUT:
MOV DL,[BX]
MOV AH, 2
INT 21H

MOV DL, 32
MOV AH, 2
INT 21H 

INC BX
LOOP OUTPUT

MAIN ENDP
