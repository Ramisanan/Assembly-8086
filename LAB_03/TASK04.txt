.MODEL SMALL

PRINTMSG MACRO MSG
PUSH AX
PUSH BX
PUSH CX
PUSH DX

LEA DX, MSG
MOV AH, 09h
INT 21H

POP DX
POP CX 
POP BX
POP AX

ENDM

.DATA
A DB 0AH, 0DH, "1st side:$"
B DB 0AH, 0DH, "2nd side:$"
C DB 0AH, 0DH, "3rd side::$"
D DB 0AH, 0DH, "Y$"
E DB 0AH, 0DH, "N$"

NUM DW 0 

X DW 0
Y DW 0
Z DW 0

.CODE
MAIN PROC
        
MOV AX, @DATA
MOV DS, AX
 
 
;START CODE HERE 

PRINTMSG A   
CALL READNUM
MOV AX, NUM
MOV X, AX
  
PRINTMSG B 
CALL READNUM
MOV BX, NUM
MOV Y, BX
  
PRINTMSG C 
CALL READNUM
MOV CX, NUM
MOV Z, CX
  
ADD AX, BX
CMP AX, CX
JL X1
  
ADD BX, CX
CMP BX ,X
JL X1
  
ADD CX, X
CMP BX, CX
JL X1   
PRINTMSG D
JMP EX
  
  
  
X1:
PRINTMSG E
EX:
MOV AH,4CH
INT 21H

READNUM proc near
PUSH AX
PUSH BX
PUSH CX
PUSH DX
MOV NUM, 00

r1:
MOV AH,01H
INT 21h
CMP AL,0DH
JE R2
MOV CX, AX
AND CX, 00FFH
SUB CX, 30H
MOV BX, 10
MOV AX, NUM
MUL BX
ADD AX, CX
MOV NUM, AX
JMP R1

R2:
POP DX
POP CX 
POP BX
POP AX
RET
READNUM ENDP

;END CODE HERE 
MOV AX, 4C00H
INT 21h
 
END MAIN
